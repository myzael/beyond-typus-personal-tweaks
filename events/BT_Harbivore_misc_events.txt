country_event = {                         #Start of event chain to cede english provinces in normandy and gascony in the HYW
	id = bt_harb_misc.6
	desc = bt_harb_misc.6.desc
	title = bt_harb_misc.6.title
	picture = CONQUEST_eventPicture
	fire_only_once = yes
	major = yes
	
	trigger = {	# Is france in HYW and occupies all english provinces in france except calais
		tag = FRA
		is_in_war = {
			casus_belli = cb_hundred_years_war
		}
		ENG = {
			france_region = {
				if = {
					limit = {
						AND = {
							is_capital = no
						}
						NOT = { province_id = 3110 }
						NOT = { province_id = 87 }
						owner = {
							OR = {
								tag = ENG
								is_subject_of = ENG
							}
						}
					}
					local_autonomy = 20
					OR = {
						controlled_by = FRA
						controller = {
							is_in_war = {
								casus_belli = cb_hundred_years_war
								NOT = {
									tag = ENG
									is_subject_of = ENG
								}
							}
						}
					}
				}
			}
		}
		war_score_against = {
			who = ENG
			value = 50
		}
	}
	
	# Cede all of Gascony and Normandy to France
	option = {
		name = bt_harb_misc.6.a
		custom_tooltip = bt_harb_misc.6.a.tt
		white_peace = ENG
		ENG = {
			country_event = {
				id = bt_harb_misc.7
			}
		}
		france_region = {
			limit = {
				NOT = { province_id = 3110 }
				NOT = { province_id = 87 }
				owner = {
					OR = {
						tag = ENG
						is_subject_of = ENG
					}
				}
			}
			cede_province = FRA
		}
		ai_chance = {
			factor = 4
		}
	}

	option = {
		name = bt_harb_misc.6.b
		custom_tooltip = bt_harb_misc.6.b.tt

		}

					}

					is_core = NRM
				}

		}
		france_region = {
			limit = {

				owner = {
					OR = {
						tag = ENG
						is_subject_of = ENG
					}
				}
				is_core = NRM
			}
			cede_province = NRM
		}
		create_subject = {
			subject_type = appanage
			subject = NRM
		}
		change_estate_land_share = {
			estate = estate_nobles
			share = -8
		}

			factor = 1
