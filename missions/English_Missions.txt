eng_french_conquest = {

	slot = 5
	generic = no
	ai = yes
	
	potential_on_load = {
		NOT = { has_dlc = "Domination" }
	}
	potential = {
		NOT = { has_dlc = "Domination" }
		OR = {
			tag = ENG
			tag = GBR
		}
		NOT = { map_setup = map_setup_random }
	}
	has_country_shield = yes
	
	eng_war_france = {
		icon = mission_conquer_50_development
		required_missions = { }
		position = 1
		
		trigger = {
			if = {
				limit = {
					exists = FRA
				}
				OR = {
					war_with = FRA
					custom_trigger_tooltip = {
						tooltip = eng_war_france.tooltip
						has_country_flag = rb_gave_up_maine
					}
					FRA = { is_subject_of = ROOT }
				}
			}
			else_if = {
				limit = { 
					NOT = { exists = FRA }
				}
				OR = {
					183 = {
						owner = {
							war_with = ROOT
						}
					}
					183 = {
						owned_by = ROOT
						is_core = ROOT
					}
				}
			}
		}
		
		effect = {
			give_permaclaims_for_mission = { mission=vassalize_france }
		}
	}
	
	eng_reclaim_arm_paris = {
		icon = mission_assemble_an_army
		required_missions = { eng_war_france }
		position = 3
		
		provinces_to_highlight = {
			OR = {
				AND = {
					owned_by = FRA
					is_capital = yes
				}
				AND = {
					province_id = 183
					NOT = { owned_by = FRA }
					owner = {
						NOT = { exists = FRA }
					}
				}
			}
			NOT = { controlled_by = ROOT }
		}
		
		trigger = {
			if = {
				limit = {
					exists = FRA
					NOT = { FRA = { is_subject_of = ROOT } }
				}
				FRA = {
					capital_scope = {
						OR = {
							controlled_by = ROOT
							AND = {
								controller = {
									OR = {
										is_subject_of = ROOT
										alliance_with = ROOT
										overlord = {
											alliance_with = ROOT
										}
									}
									custom_trigger_tooltip = {
										tooltip = joint_war_with_root_vs_france
										OR = {
											is_in_war = {
												attackers = root
												attackers = this
												defenders = FRA
											}
											is_in_war = {
												defenders = root
												defenders = this
												attackers = FRA
											}
										}
									}
								}
							}
						}
					}
				}
			}
			else_if = {
				limit = {
					exists = FRA
					FRA = { is_subject_of = ROOT }
				}
				FRA = { is_subject_of = ROOT }
			}
			else = {
				183 = {
					owned_by = ROOT
				}
			}
		}
		
		effect = {
			if = {
				limit = {
					FRA = {
						is_subject = no
					}
				}
				add_casus_belli = {
					target = FRA
					type = cb_hundred_years_union
					months = 360
				}
			}
			if = {
				limit = {
					has_dlc = "Rule Britannia"
				}
				give_permaclaims_for_mission = { mission=eng_rb_brittany }
			}
		}
	}
	
	vassalize_france = {
		icon = mission_monarch_in_throne_room
		required_missions = { eng_reclaim_arm_paris }
		position = 5
		
		provinces_to_highlight = {
			province_group = vassalize_france_provinces
			NOT = { country_or_non_sovereign_subject_holds = ROOT }
		}
		
		trigger = {
			has_fulfiled_province_requirement_for_mission_trigger = {
				tooltip=country_holds_or_non_sovereign_subject_holds_highlighted 
				provinces=vassalize_france_provinces
				owner_type=country_or_non_sovereign_subject_holds
			}
		}
		
		effect = {
			if = {
				limit = {
					has_heir = yes
					has_ruler_flag = henry_vi_lancaster
					NOT = { has_country_flag = rb_gave_up_maine }
				}
				change_adm = 1
				change_dip = 1
				change_mil = 1
			}
			add_prestige = 25
			add_country_modifier = {
				name = "eng_french_recovery"
				duration = 5475
			}
		}
	}
}
